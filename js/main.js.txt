// main.js
import { db } from "./firebase-config.js";
import { collection, getDocs } from "https://www.gstatic.com/firebasejs/9.0.0/firebase-firestore.js";

async function loadVendors() {
  const vendorListEl = document.getElementById("vendor-list");
  try {
    const vendorsSnapshot = await getDocs(collection(db, "vendors"));
    vendorsSnapshot.forEach(doc => {
      const vendor = doc.data();
      const vendorEl = document.createElement("div");
      vendorEl.className = "vendor";
      vendorEl.innerHTML = `
        <h3>${vendor.name}</h3>
        <p>${vendor.description}</p>
        <a href="vendor.html?id=${doc.id}">Visit Store</a>
      `;
      vendorListEl.appendChild(vendorEl);
    });
  } catch (error) {
    console.error("Error fetching vendors:", error);
    vendorListEl.innerHTML = "<p>Failed to load vendors. Please try again later.</p>";
  }
}

document.addEventListener("DOMContentLoaded", loadVendors);
